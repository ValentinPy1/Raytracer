50 template <typename T>
51 T *loadInstance(const std::string &path, const std::string &libName) {
52 void *handle = dlopen(path.c_str(), RTLD_LAZY);
53 if (!handle) {
54 throw DLLoaderException("Cannot open library: " + std::string(dlerror()));
55 }
56 dlerror();
57 T *instance = reinterpret_cast<T *>(dlsym(handle, "entryPoint"));
58 const char *dlsym_error = dlerror();
59 if (dlsym_error) {
60 throw DLLoaderException("Cannot load symbol 'entryPoint': " + std::string(dlsym_error));
61 }
62 _handles.push_back(handle);
63 _handleNames.push_back(libName);
64 return instance;
65 }

77 template <typename T>
78 T loadSymbol(const std::string &libName, const std::string &symbolName) {
79 for (size_t i = 0; i < _handleNames.size(); i++) {
80 if (_handleNames[i] == libName) {
81 T sym = reinterpret_cast<T>(dlsym(_handles[i], symbolName.c_str()));
82 const char *dlsym_error = dlerror();
83 if (dlsym_error) {
84 throw DLLoaderException("Cannot load symbol '" + symbolName + "': " + std::string(dlsym_error));
85 }
86 return sym;
87 }
88 }
89 throw DLLoaderException("Cannot find library " + libName);
90 }

98 std::vector<void *> _handles;
