52 T *loadInstance(const std::string &libName, const std::string &path = "") {
53 void *handle = nullptr;
54 if (_libHandles.find(libName) != _libHandles.end()) {
55 handle = _libHandles[libName];
56 } else {
57 handle = dlopen(path.c_str(), RTLD_LAZY);
58 if (!handle) {
59 throw DLLoaderException("Cannot open library: " + std::string(dlerror()));
60 }
61 dlerror();
62 _libHandles[libName] = handle;
63 }
64 try {
65 T *(*entryPoint)() = reinterpret_cast<T *(*)()>(dlsym(handle, "entryPoint"));
66 const char *dlsym_error = dlerror();
67 if (dlsym_error) {
68 throw DLLoaderException("Cannot load symbol 'entryPoint': " + std::string(dlsym_error));
69 }
70 return entryPoint();
71 } catch (const std::exception &e) {
72 throw DLLoaderException("Cannot load symbol 'entryPoint': " + std::string(e.what()));
73 }
74 }

86 template <typename T>
87 T loadSymbol(const std::string &libName, const std::string &symbolName) {
88 if (_libHandles.find(libName) == _libHandles.end()) {
89 throw DLLoaderException("Cannot find library " + libName);
90 }
91 try {
92 T sym = reinterpret_cast<T>(dlsym(_libHandles[libName], symbolName.c_str()));
93 const char *dlsym_error = dlerror();
94 if (dlsym_error) {
95 throw DLLoaderException("Cannot load symbol '" + symbolName + "': " + std::string(dlsym_error));
96 }
97 return sym;
98 } catch (const std::exception &e) {
99 throw DLLoaderException("Cannot load symbol '" + symbolName + "': " + std::string(e.what()));
100 }
101 }
