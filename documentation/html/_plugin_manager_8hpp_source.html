54 T *loadInstance(const std::string &libName, const std::string &path = "") {
55 void *handle = nullptr;
56 if (_libHandles.find(libName) != _libHandles.end()) {
57 handle = _libHandles[libName];
58 } else {
59 handle = dlopen(path.c_str(), RTLD_LAZY);
60 if (!handle) {
61 throw DLLoaderException("Cannot open library: " + std::string(dlerror()));
62 }
63 dlerror();
64 _libHandles[libName] = handle;
65 }
66 try {
67 T *(*entryPoint)() = reinterpret_cast<T *(*)()>(dlsym(handle, "entryPoint"));
68 const char *dlsym_error = dlerror();
69 if (dlsym_error) {
70 throw DLLoaderException("Cannot load symbol 'entryPoint': " + std::string(dlsym_error));
71 }
72 return entryPoint();
73 } catch (const std::exception &e) {
74 throw DLLoaderException("Cannot load symbol 'entryPoint': " + std::string(e.what()));
75 }
